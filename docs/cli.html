Handle command line options

 
argv = optimist
  .usage('Usage: $0')
  .options('h',
    alias     : 'help'
    boolean   : true
    describe  : 'Show this help info and exit'
  )
  .options('o',
    alias     : 'host'
    default   : ''
    describe  : 'Host to accept connections on, false == any'
  )
  .options('u',
    alias     : 'url'
    default   : ''
    describe  : 'Url to be used for the realm in openID'
  )
  .options('p',
    alias     : 'port'
    default   : 3000
    describe  : 'Port'
  )
  .options('r',
    alias     : 'root'
    default   : path.join(__dirname)
    describe  : 'Application root folder'
  )
  .options('g',
    alias     : 'pdfgen-url'
    default   : ''
    describe  : 'URL to the PDF generation service (accepts /deposit)'
  )
  .options('x',
    alias     : 'debug-user'
    boolean   : true
    describe  : 'Set this flag if you do not want to have to authenticate'
  )
  .options('test',
    boolean   : true
    describe  : 'Set server to work with the rspec integration tests'
  )
  .argv
